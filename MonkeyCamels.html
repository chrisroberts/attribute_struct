def camel_initialize_copy(orig, hump=nil)
  new_val = un_camel_initialize_copy(orig)
  if(hump.nil?)
    orig._camel? ? new_val : new_val._no_hump
  else
    new_val._no_hump if hump == false
  end
end

def camel_to_s
